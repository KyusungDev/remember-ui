## Basic usage

<Playground>
  {() => {
    const [isChecked, setIsChecked] = React.useState(true);
    const toggle = () => setIsChecked(!isChecked);
    return (
      <>
        <DoczContainer direction="column">
          <div>체크박스 on/off/disabled( Static )</div>
          <DoczContainer direction="row">
            <Checkbox state="on" />
            <Checkbox state="off" />
            <Checkbox state="on" disabled />
          </DoczContainer>
        </DoczContainer>
        
        <DoczContainer direction="column">
          <div>체크박스 onClick Event Bind( Dynamic )</div>
          <DoczContainer direction="row">
            <Checkbox state={isChecked ? 'on' : 'off'} onClick={toggle} />
            <Checkbox
              state={isChecked ? 'on' : 'off'}
              onClick={toggle}
              disabled
            />
          </DoczContainer>
        </DoczContainer>
      </>
    );
  }}
</Playground>
