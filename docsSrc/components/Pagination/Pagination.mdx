## Basic usage

<Playground>
  {() => {
    const [pageNumber1, setPageNumber1] = React.useState(1);
    const [pageNumber2, setPageNumber2] = React.useState(2);
    return (
      <DoczContainer direction="column">
        <Pagination
          onChangePage={page => {
            setPageNumber1(page);
          }}
          activePage={pageNumber1}
          totalItemCount={56}
          pageRangeDisplayed={10}
          itemCountPerPage={10}
        />
        <Pagination
          onChangePage={page => {
            setPageNumber2(page);
          }}
          activePage={pageNumber2}
          totalItemCount={56}
          pageRangeDisplayed={2}
          itemCountPerPage={10}
        />
      </DoczContainer>
    );
  }}
</Playground>
